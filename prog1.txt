// Fibonacci Series using Recursion
import java.io.*;
class fibonacci {
	static int fib(int num)
	{
		if (num <= 1)
			return num;
		return fib(num - 1) + fib(num - 2);
	}

	public static void main(String args[])
	{
		int n = 9;
		System.out.println(fib(n));
	}
}
