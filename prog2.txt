// Fibonacci Series using Recursion
import java.io.*;
class FibTantan {
	static int fib(int number)
	{
		if (number <= 1)
			return number;
		return fib(number - 1) + fib(number - 2);
	}

	public static void main(String args[])
	{
		int number = 9;
		System.out.println(fib(number));
	}
}
